#Kullanýcýdan girdiði iki sayý ve yapýlacak iþlem türü (toplama, çýkarma, çarpma, bölme) seçildiðinde, sonucu hesaplayarak ekranda gösteren programý yazalým.

sayi1=int(input("1. sayýyý GÝRÝNÝZ:"))
sayi2=int(input("2. sayýyý GÝRÝNÝZ:"))

print("Toplama=1, Çýkarma=2, Çarpma=3, Bölme=4")
islem=int(input("Yapmak istediðiniz iþlem nedir?"))

if islem==1:
    print("CEVAP: "+str(sayi1+sayi2))
if islem==2:
    print("CEVAP: "+str(sayi1-sayi2))
if islem==3:
    print("CEVAP: "+str(sayi1*sayi2))
if islem==4:
    print("CEVAP: "+str(sayi1/sayi2))
