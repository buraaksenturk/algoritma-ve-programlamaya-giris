elma=int(input("1. sayýyý GÝRÝNÝZ:"))
armut=int(input("2. sayýyý GÝRÝNÝZ:"))
ayva=int(input("3. sayýyý GÝRÝNÝZ:"))

    kucuk=elma

    kucuk=armut
